\begin{ask}
 Να αποδειχθούν οι παρακάτω τριγωνομετρικές ταυτότητες:\vskip\vert
 \begin{enumerate}
 \item \(\dfrac{1-\ef ^2 x}{1+\ef ^2 x}=1-2\hm ^2x \)
 \item \(\dfrac{1+\hm ^2x}{\syn ^2 x}-2\ef ^2x =1\)
 \item \( \dfrac{1}{\hm^2 \omega \cdot \syn ^2 \omega}-\dfrac{1}{\syn ^2 \omega}=\dfrac{1}{\hm ^2 \omega } \)
 \item \( \dfrac{1}{\syn^2 \phi}-\dfrac{1}{\sef ^2\phi}=1\)
 \end{enumerate}
\end{ask}
\begin{ask}
 Να αποδειχθούν οι παρακάτω τριγωνομετρικές ταυτότητες:\vskip\vert
 \begin{enumerate}
  \item \(\dfrac{1+\ef ^2 \omega}{\syn ^2 \omega}\cdot \sef ^4\omega =\dfrac{1+\sef ^2 \omega }{\hm ^2 \omega} \)
  \item \(\dfrac{1}{\hm \phi} - \dfrac{\hm \phi}{1+\syn \phi} =\sef \phi\)
  \item \( \dfrac{\ef x}{1-\sef x}+\dfrac{\sef x}{1-\ef x}-1=\dfrac{1}{\hm x \cdot \syn x} \)
  \item \(\dfrac{\syn \omega}{1+\ef \omega}-\dfrac{\hm \omega}{1+\sef \omega}=\syn \omega -\hm \omega \)
 \end{enumerate}
\end{ask}
